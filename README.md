### `Question 3`
```
const soal3 = (arr = []) => {
  let result = {}

  arr.map((word) => {
    let key = word.split("").sort().join("")
    if (result[key]) return result[key].push(word)
    return result[key] = [word]
  })

  return Object.values(result)
}
```
